DROP TABLE IF EXISTS Rental;

CREATE TABLE Rental (
RNum NUMBER(20) NOT NULL,
MID NUMBER(10) NOT NULL,
BNum VARCHAR2(20) NOT NULL,
RDDATE DATE NULL,

);

DROP TABLE IF EXISTS Book;

CREATE TABLE Book (
BNum VARCHAR2(20) NOT NULL,
Auther VARCHAR2(20) NULL,
BName VARCHAR2(20) NULL,
Publisher VARCHAR2(20) NULL,
PDATE DATE NULL,
Cl VARCHAR2(5)
);
